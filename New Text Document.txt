import streamlit as st
import pandas as pd
import pickle
import shap
import warnings
from streamlit_shap import st_shap

warnings.filterwarnings("ignore")
st.set_page_config(page_title="Loan Approval Predictor", page_icon="ðŸ¦", layout="centered")

# === Load model components ===
with open("label_encoders.pkl", "rb") as f:
    label_encoders = pickle.load(f)
    label_encoders.pop('Loan_Status', None)

with open("scaler.pkl", "rb") as f:
    scaler = pickle.load(f)

with open("X_columns.pkl", "rb") as f:
    trained_columns = pickle.load(f)

with open("model_rf.pkl", "rb") as f:
    model = pickle.load(f)

# === Collect user input ===
def user_input_features():
    st.markdown("### ðŸ“ Please enter your loan details:")

    gender = st.selectbox("Gender", label_encoders['Gender'].classes_)
    married = st.selectbox("Married", label_encoders['Married'].classes_)
    dependents = st.selectbox("Dependents", ['0', '1', '2', '3'])
    education = st.selectbox("Education", label_encoders['Education'].classes_)
    self_employed = st.selectbox("Self Employed", label_encoders['Self_Employed'].classes_)
    
    applicant_income = st.number_input("Applicant Income (â‚¹)", min_value=0)
    coapplicant_income = st.number_input("Coapplicant Income (â‚¹)", min_value=0)
    loan_amount = st.number_input("Loan Amount (â‚¹)", min_value=0)
    loan_amount_term = st.number_input("Loan Amount Term (months)", min_value=0)
    credit_history = st.selectbox("Credit History", [0.0, 1.0])
    property_area = st.selectbox("Property Area", label_encoders['Property_Area'].classes_)

    data = {
        'Gender': gender,
        'Married': married,
        'Dependents': dependents,
        'Education': education,
        'Self_Employed': self_employed,
        'ApplicantIncome': applicant_income,
        'CoapplicantIncome': coapplicant_income,
        'LoanAmount': loan_amount,
        'Loan_Amount_Term': loan_amount_term,
        'Credit_History': credit_history,
        'Property_Area': property_area
    }

    return pd.DataFrame([data]), data  # Return raw dict for display

# === Preprocess input ===
def preprocess_input(df):
    for col in label_encoders:
        if col in df.columns:
            df[col] = label_encoders[col].transform(df[col].astype(str))
    df['Dependents'] = df['Dependents'].astype(int)
    num_cols = ['ApplicantIncome', 'CoapplicantIncome', 'LoanAmount', 'Loan_Amount_Term']
    df[num_cols] = scaler.transform(df[num_cols])
    return df[trained_columns]

# === App UI ===
st.title("ðŸ¦ Loan Approval Prediction")
st.markdown("Use this tool to check if your loan is likely to be approved.")

input_df, raw_input = user_input_features()
processed_df = preprocess_input(input_df)

if st.button("ðŸ” Predict"):
    prediction = model.predict(processed_df)[0]
    prediction_proba = model.predict_proba(processed_df)[0][prediction]
    result = "âœ… Approved" if prediction == 1 else "âŒ Rejected"

    st.success(f"### Loan Status: {result}  \nConfidence: **{prediction_proba:.2f}**")

    # === SHAP Explanation ===
    st.markdown("---")
    st.markdown("### ðŸ” Why this prediction?")

    explainer = shap.TreeExplainer(model)
    shap_values = explainer.shap_values(processed_df)

    if isinstance(shap_values, list):
        shap_vals = shap_values[1]  # Class 1
    else:
        shap_vals = shap_values

    shap_row = shap_vals[0]
    if shap_row.ndim == 2 and shap_row.shape[1] == 2:
        shap_series = pd.Series(shap_row[:, 1], index=processed_df.columns)
    else:
        shap_series = pd.Series(shap_row, index=processed_df.columns)

    top_features = shap_series.abs().sort_values(ascending=False).head(3).index.tolist()

    st.markdown("#### ðŸ“Š Top contributing features:")
    for i, feat in enumerate(top_features, 1):
        st.markdown(f"- {i}. **{feat}**")

    st.markdown("#### ðŸ—£ï¸ Explanation in simple words:")

    explanation_sentences = []
    for feat in top_features:
        value = raw_input.get(feat, "N/A")
        shap_val = shap_series[feat]
        direction = "increased" if shap_val > 0 else "decreased"

        if feat == 'ApplicantIncome':
            explanation_sentences.append(f"Your income of â‚¹{value} {direction} the chances of loan approval.")
        elif feat == 'CoapplicantIncome':
            explanation_sentences.append(f"Your coapplicant's income of â‚¹{value} {direction} the approval likelihood.")
        elif feat == 'LoanAmount':
            explanation_sentences.append(f"The requested loan amount of â‚¹{value} {direction} the risk.")
        elif feat == 'Credit_History':
            if shap_val > 0:
                explanation_sentences.append("Your credit history helped your chances.")
            else:
                explanation_sentences.append("Lack of good credit history reduced your chances.")
        elif feat == 'Loan_Amount_Term':
            explanation_sentences.append(f"The loan term of {value} months {direction} the likelihood of approval.")
        elif feat == 'Education':
            explanation_sentences.append("Your education status influenced the decision.")
        elif feat == 'Self_Employed':
            explanation_sentences.append("Being self-employed impacted your approval chances.")
        elif feat == 'Property_Area':
            explanation_sentences.append("The area of the property played a role in the decision.")
        else:
            explanation_sentences.append(f"{feat} {direction} your chances.")

    for sentence in explanation_sentences:
        st.markdown(f"- {sentence}")
        st.markdown("#### ðŸ”¬ SHAP Force Plot:")

    # Generate SHAP force plot for the current input
    shap.initjs()
    force_plot = shap.force_plot(
        explainer.expected_value[1],  # for class 1
        shap_values[1][0],            # first row of class 1 SHAP values
        processed_df.iloc[0],
        matplotlib=False,
        feature_names=processed_df.columns
    )

    st_shap(force_plot, height=300)
